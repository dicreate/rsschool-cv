Human readable duration format 

function formatDuration (seconds) {

	let result = []
	let years = Math.floor(seconds / 31536000)
	let days = Math.floor((seconds - years * 31536000) / 86400)
	let hours =  Math.floor((seconds - years * 31536000 - days * 86400) / 3600)
	let minutes = Math.floor((seconds - years * 31536000 - days * 86400 - hours * 3600) / 60)
	let sec = (seconds - years * 31536000 - days * 86400 - hours * 3600 - minutes * 60)
	let clock = [years,days,hours, minutes, sec]
	
	if(seconds == 0){
		return 'now'
	}
	
	if(clock[0] > 0){
	clock[0] = clock[0] > 1 ? clock[0] + ' years' : clock[0] + ' year'
	result.push(clock[0])
	}
	
	if(clock[1] > 0){
	clock[1] = clock[1] > 1 ? clock[1] + ' days' : clock[1] + ' day'
	result.push(clock[1])
	}
	
	if(clock[2] > 0){
	clock[2] = clock[2] > 1 ? clock[2] + ' hours' : clock[2] + ' hour'
	result.push(clock[2])
	}
	
	if(clock[3] > 0){
	clock[3] = clock[3] > 1 ? clock[3] + ' minutes' : clock[3] + ' minute'
	result.push(clock[3])
	}
	
	if(clock[4] > 0){
	clock[4] = clock[4] > 1 ? clock[4] + ' seconds' : clock[4] + ' second'
	result.push(clock[4])
	}
	
	if(result.length > 1){
		result = result.map(el => el == result[result.length - 1] || el == result[result.length - 2] ? el : el + ",")
		result[result.length - 1] = "and " + result[result.length - 1]
	}
	
	return result.join(" ")
	
	}
